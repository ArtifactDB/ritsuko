60inline Version parse_version_string(const char* version_string, size_t size, bool skip_patch = false) {
61 int major = 0, minor = 0, patch = 0;
62 size_t i = 0, end = size;
63
64 // MAJOR VERSION.
65 if (size == 0) {
66 throw std::runtime_error("version string is empty");
67 }
68 if (version_string[i] == '0') {
69 ++i;
70 if (i < end && version_string[i] != '.') {
71 throw_version_error(version_string, size, "has leading zeros in its major version");
72 }
73 } else {
74 while (i < end && version_string[i] != '.') {
75 if (!std::isdigit(version_string[i])) {
76 throw_version_error(version_string, size, "contains non-digit characters");
77 }
78 major *= 10;
79 major += version_string[i] - '0';
80 ++i;
81 }
82 }
83
84 // MINOR VERSION.
85 if (i == end) {
86 throw_version_error(version_string, size, "is missing a minor version");
87 }
88 ++i; // get past the period and check again.
89 if (i == end) {
90 throw_version_error(version_string, size, "is missing a minor version");
91 }
92
93 if (version_string[i] != '0') {
94 while (i < end && version_string[i] != '.') {
95 if (!std::isdigit(version_string[i])) {
96 throw_version_error(version_string, size, "contains non-digit characters");
97 }
98 minor *= 10;
99 minor += version_string[i] - '0';
100 ++i;
101 }
102
103 } else {
104 ++i;
105 if (i < end && version_string[i] != '.') {
106 throw_version_error(version_string, size, "has leading zeros in its minor version");
107 }
108 }
109
110 if (skip_patch) {
111 if (i != end) {
112 throw_version_error(version_string, size, "should not have a patch version");
113 }
114 return Version(major, minor, 0);
115 }
116
117 // PATCH VERSION.
118 if (i == end) {
119 throw_version_error(version_string, size, "is missing a patch version");
120 }
121 ++i; // get past the period and check again.
122 if (i == end) {
123 throw_version_error(version_string, size, "is missing a patch version");
124 }
125
126 if (version_string[i] == '0' && i + 1 < end) {
127 throw_version_error(version_string, size, "has leading zeros in its patch version");
128 }
129 while (i < end) {

131 throw_version_error(version_string, size, "contains non-digit characters");
132 }
133 patch *= 10;
134 patch += version_string[i] - '0';

136 }
