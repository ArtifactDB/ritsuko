37inline std::vector<std::string> load_1d_string_attribute(const H5::Attribute& attr, hsize_t full_length) {
38 auto dtype = attr.getDataType();
39 auto mspace = attr.getSpace();
40 std::vector<std::string> output;
41 output.reserve(full_length);
42
43 if (dtype.isVariableStr()) {
44 std::vector<char*> buffer(full_length);
45 try {
46 attr.read(dtype, buffer.data());
47 for (hsize_t i = 0; i < full_length; ++i) {
48 output.emplace_back(buffer[i]);
49 }
50 } catch (...) {
51 H5Dvlen_reclaim(dtype.getId(), mspace.getId(), H5P_DEFAULT, buffer.data());
52 throw;
53 }
54
55 } else {
56 size_t len = dtype.getSize();
57 std::vector<char> buffer(len * full_length);
58 attr.read(dtype, buffer.data());
59 auto ptr = buffer.data();
60 for (size_t i = 0; i < full_length; ++i, ptr += len) {
61 size_t j = 0;
62 for (; j < len && ptr[j] != '\0'; ++j) {}
63 output.emplace_back(ptr, ptr + j);
64 }
65 }
66
67 return output;
68}
